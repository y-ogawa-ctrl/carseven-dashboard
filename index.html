function doGet(e) { let result; try { const dataset = (e && e.parameter && e.parameter.dataset) || 'weekly'; if (dataset === 'sales') { result = { ok: true, data: getSalesData(), generatedAt: new Date().toISOString() }; } function cellToStr(val) { if (val === null || val === undefined || val === '') return ''; if (val instanceof Date) { return (val.getMonth() + 1) + '月' + val.getDate() + '日'; } return String(val).trim(); } function findCol(header, keyword) { for (let i = 0; i < header.length; i++) { if (String(header[i]).includes(keyword)) return i; } return -1; } // ── 週次MTGデータ取得（既存） ────────────────────────────── function getAllData() { const rows = []; for (const storeName in STORE_FILE_IDS) { const fileId = STORE_FILE_IDS[storeName]; try { const ss = SpreadsheetApp.openById(fileId); for (const sheet of ss.getSheets()) { const name = sheet.getName(); if (!/^2[56]\d{2,3}$/.test(name)) continue; const data = sheet.getDataRange().getValues(); if (data.length < 2) continue; let hIdx = -1; for (let i = 0; i < Math.min(6, data.length); i++) { const row = data[i]; for (let j = 0; j < row.length; j++) { if (String(row[j]).includes('顧客名')) { hIdx = i; break; } } if (hIdx >= 0) break; } if (hIdx < 0) continue; const hdr = data[hIdx]; const C = { rcv: findCol(hdr, '車両受領日'), cust: findCol(hdr, '顧客名'), car: findCol(hdr, '車名'), type: findCol(hdr, '車種区分'), meik: findCol(hdr, '名変完了日'), doc: findCol(hdr, '書類の状況'), taxProc: findCol(hdr, '自税還付手続き対応日'), taxDate: findCol(hdr, '自税還付日'), tant: findCol(hdr, '仕入担当'), payChk: findCol(hdr, '支払完了したら'), }; for (let i = hIdx + 1; i < data.length; i++) { const r = data[i]; const cust = cellToStr(r[C.cust]); const rcv = cellToStr(r[C.rcv]); if (!cust && !rcv) continue; rows.push({ store: storeName, month: name, cust: cust, car: cellToStr(r[C.car]), type: cellToStr(r[C.type]), tant: cellToStr(r[C.tant]), rcv: rcv, meik: cellToStr(r[C.meik]), doc: cellToStr(r[C.doc]), taxProc: cellToStr(r[C.taxProc]), taxDate: cellToStr(r[C.taxDate]), payChk: C.payChk >= 0 ? cellToStr(r[C.payChk]) : null, }); } } } catch(e) {} } return rows; } // ── 販売納車管理データ取得（新規） ────────────────────────── function getSalesData() { const rows = []; const ss = SpreadsheetApp.openById(SALES_FILE_ID); for (const sheet of ss.getSheets()) { const sheetName = sheet.getName(); if (data.length < 2) continue; // ヘッダー行を探す（「契約日」を含む行） let hIdx = -1; for (let i = 0; i < Math.min(10, data.length); i++) { const row = data[i]; for (let j = 0; j < row.length; j++) { if (String(row[j]).includes('契約日')) { hIdx = i; break; } } if (hIdx >= 0) break; } if (hIdx < 0) continue; const hdr = data[hIdx]; const C = { keiyaku: findCol(hdr, '契約日'), cust: findCol(hdr, 'お客様名'), car: findCol(hdr, '車名'), tant: findCol(hdr, '担当者'), // 整備工場の車両出庫日 seibiOut: findColNth(hdr, '車両', 0), // 登録依頼日 reqDate: findCol(hdr, '登録'), // 入金OK nyukinOk: findCol(hdr, '入金OK'), // 納車完了日 delivered: findCol(hdr, '納車'), }; // 列の解析: ヘッダーが複雑なので、近接列で詳細特定 // 「車両出庫日」は「整備工場」セクションの中にある // → 「整備依頼日」の次が「車両出庫日」 const seibiReqIdx = findColNthMulti(hdr, '整備', '依頼日'); const carOutIdx = seibiReqIdx >= 0 ? seibiReqIdx + 1 : -1; // 「登録依頼日」は最初に出てくる const regReqIdx = findColMulti(hdr, '登録', '依頼日'); const cust = cellToStr(r[C.cust]); // 契約日も顧客名もなければスキップ if (!keiyaku && !cust) continue; // *1, *2, *3 のような行はスキップ if (String(r[0]).includes('*')) continue; rows.push({ store: storeName, cust: cust, car: cellToStr(r[C.car]), tant: cellToStr(r[C.tant]), keiyaku: keiyaku, carOut: carOutIdx >= 0 ? cellToStr(r[carOutIdx]) : '', regReq: regReqIdx >= 0 ? cellToStr(r[regReqIdx]) : '', payOk: payOkIdx >= 0 ? cellToStr(r[payOkIdx]) : '', delivered: delIdx >= 0 ? cellToStr(r[delIdx]) : '', }); count++; } } return -1; } // 補助関数: 2つのキーワード両方を含む列 function findColMulti(header, kw1, kw2) { for (let i = 0; i < header.length; i++) { const v = String(header[i]); if (v.includes(kw1) && v.includes(kw2)) return i; } return -1; } // 補助関数: 完全一致または、ほぼ一致 function findColExact(header, keyword) { for (let i = 0; i < header.length; i++) { if (String(header[i]).trim() === keyword || String(header[i]).includes(keyword)) return i; } return -1; } // 補助関数: 「整備工場」セクション内の「依頼日」 function findColNthMulti(header, sectionKw, fieldKw) { // 全セルから検索（セクション情報がヘッダーに含まれる場合） for (let i = 0; i < header.length; i++) { const v = String(header[i]);
